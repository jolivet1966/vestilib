    </div>

          <div className="w-16 h-16 bg-[#F5C84A] rounded-full flex items-center justify-center text-[#1A3A6B] text-xl font-black">
            {initiales}
          </div>
          <div>
            <p className="text-white font-bold text-lg">{host.prenom} {host.nom}</p>
            <p className="text-white/50 text-sm">{host.email}</p>
            <p className="text-white/50 text-xs">{host.ville}</p>
          </div>
        </div>
      </div>

      {/* Tabs */}
      <div className="bg-white border-b border-gray-100 sticky top-0 z-10">

            Ã€ propos

            className={`flex-1 py-3 text-sm font-medium border-b-2 transition-colors ${menu === 'compte' ? 'border-[#1A3A6B] text-[#1A3A6B]' : 'border-transparent text-gray-400'}`}>

                <div className="text-center bg-green-50 rounded-xl p-3">
                  <p className="text-xl font-black text-green-600">{totalGagne.toFixed(0)}â‚¬</p>
                  <p className="text-xs text-gray-400 mt-1">Total gagnÃ©</p>
                </div>

                  <p className="text-xl font-black text-[#1A3A6B]">{balance?.available?.toFixed(0) ?? 'â€”'}â‚¬</p>

                    {savingInfo ? 'Sauvegarde...' : 'Enregistrer'}
                  </button>
                </div>
              ) : (
                <div className="space-y-2 text-sm">
                  <Row label="Nom" value={`${host.prenom} ${host.nom}`} />
                  <Row label="Email" value={host.email} />
                  <Row label="TÃ©lÃ©phone" value={host.telephone || 'â€”'} />

            </div>

            {/* Validation profil */}
            <div className="bg-white rounded-2xl border border-gray-100 p-5 shadow-sm">

                <ValidationRow label="PiÃ¨ce d'identitÃ©" status={host.stripePayoutsEnabled} info="ValidÃ©e via Stripe" />
                <ValidationRow label="Email validÃ©" status={true} info={host.email} />
                <ValidationRow label="TÃ©lÃ©phone validÃ©" status={!!host.telephone} info={host.telephone || 'Non renseignÃ©'} />
              </div>
            </div>

            {/* Rappel capacitÃ©s */}
            <div className="bg-white rounded-2xl border border-gray-100 p-5 shadow-sm">
              <p className="text-xs font-semibold text-gray-400 uppercase tracking-wider mb-4">CapacitÃ©s configurÃ©es</p>

                <Row label="ðŸŽ’ Max articles consigne" value={host.capaciteMax ? `${host.capaciteMax} articles` : 'Non dÃ©fini'} />
                <Row label="ðŸï¸ Max parking moto" value={host.capaciteMaxMoto ? `${host.capaciteMaxMoto} motos` : 'Non dÃ©fini'} />
                <Row label="ðŸš² Max parking vÃ©lo" value={host.capaciteMaxVelo ? `${host.capaciteMaxVelo} vÃ©los` : 'Non dÃ©fini'} />
              </div>
            </div>
          </>
        )}

        {/* â”€â”€ MENU 2 : COMPTE â”€â”€ */}
        {menu === 'compte' && (

            <MenuCard icon="â­" title="Avis" subtitle="Vos Ã©valuations clients" onClick={() => {}} />

            <div className="bg-white rounded-2xl border border-gray-100 shadow-sm overflow-hidden">
              <button onClick={() => setShowPwd(!showPwd)}
                className="w-full flex items-center justify-between p-5 hover:bg-gray-50 transition-colors">
                <div className="flex items-center gap-3">
                  <span className="text-xl">ðŸ”‘</span>
                  <div className="text-left">
                    <p className="text-sm font-medium text-gray-800">Mot de passe</p>

                  </div>
                </div>

                  <Field label="Nouveau mot de passe" value={newPwd} onChange={setNewPwd} type="password" />

                    className="w-full bg-[#1A3A6B] text-[#F5C84A] font-medium py-2.5 rounded-xl hover:bg-[#0C2447] disabled:opacity-50 transition-colors text-sm">

                      </span>

            <div className="bg-white rounded-2xl border border-red-100 shadow-sm overflow-hidden">

                className="w-full flex items-center gap-3 p-4 hover:bg-red-50 transition-colors">

function Field({ label, value, onChange, type = 'text' }: { label: string; value: string; onChange: (v: string) => void; type?: string }) {
  return (
    <div>
      <label className="text-xs text-gray-500 block mb-1">{label}</label>
      <input type={type} value={value} onChange={e => onChange(e.target.value)}
        className="w-full border border-gray-200 rounded-xl px-3 py-2.5 text-sm focus:outline-none focus:border-[#1A3A6B] transition-colors" />
    </div>
  )
}

function Row({ label, value }: { label: string; value: string }) {
  return (
    <div className="flex justify-between items-center py-1.5 border-b border-gray-50 last:border-0">
      <span className="text-gray-400 text-xs">{label}</span>
      <span className="text-gray-700 text-xs font-medium text-right max-w-[60%]">{value}</span>
    </div>
  )
}

function ValidationRow({ label, status, info }: { label: string; status: boolean; info: string }) {

}

function MenuCard({ icon, title, subtitle, onClick }: { icon: string; title: string; subtitle: string; onClick: () => void }) {

  )
}
